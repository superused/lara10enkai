# OpenSSH serverのインストール 
sudo apt-get install openssh-server

# 秘密鍵の作成
ssh-keygen

# Enter file in which to save the key (/home/ubuntu/.ssh/id_rsa):
エンターキーを押す

# Enter passphrase (empty for no passphrase):
エンターキーを押す（入れたかったら自由にどうぞ）
もう一回エンターキーを押す

# 鍵のconfigの設定
vim ~/.ssh/config

# 内容は以下で保存
Host github.com
HostName github.com
User git
IdentityFile "/home/ubuntu/.ssh/id_rsa"

# 鍵をgithubに登録
https://github.com/settings/keys にアクセス
New SSH Key をクリック
Title 適当
Key　以下を入力
cat ~/.ssh/id_rsa.pub　の結果で出た内容を入力

Add SSH Keyボタンを押して登録


# githubとの接続の確認
ssh -T git@github.com

# Are you sure you want to continue connecting (yes/no/[fingerprint])?
yes

# 以下のような表示が出たらOK
Hi taichi-nakasone! You've successfully authenticated, but GitHub does not provide shell access.


# pushするリポジトリを設定
git remote add origin git@github.com:taichi-nakasone/lara8order.git (初回)
git remote set-url origin git@github.com:taichi-nakasone/lara8order.git (変更したい場合)

git remote -v のコマンドで確認


git コマンド集
#shimojoというbranchを送る
git pull origin shimojo

#今いるブランチをshimojo2に相続させる形でブランチを作成
git branch shimojo2

#ブランチをコミットするときに使う
tig
s:sutatus
(statusで)Shift-c:コミット

#ブランチのひとつ前(^の回数分だけ戻る)の動作を復元
git reset --hard HEAD^

#今いるブランチとshimojo2のブランチをマージ
git merge shimojo2

#branchの移動
git checkout shimojo

#接続サーバーのブランチを削除
git push -d origin branch名

#shimojoブランチの変更をサーバーに送る
git push origin shimojo

#tigを押すとnanoがデフォルトで開くのでvimに変更
git config --global core.editor "vim"

#main(master)ブランチに移動
git checkout main	

#コミット
git commit -m "laravelのコミット"

#現在のブランチのステータス(ファイル)の確認
git status

#現在のファイルをgitサーバーに登録
git add .

#gitでサーバーに送るファイルのうち(重いファイルなど)無視するファイルのリストを確認
less .gitignore

#現在のディレクトリでgitサーバーに送るファイルの一覧を表示
git log -p
